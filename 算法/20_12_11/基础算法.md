# 基础算法（一）

## 快速排序

> 1. 确定分界点
> 2. 划分区间：左侧 <= x、右侧 >= x
> 3. 递归处理左右两侧

### 快排模版

```C++
void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j), quick_sort(q, j + 1, r);
}
```

### AcWing 785. 快速排序

```C++
#include <iostream>

using namespace std;

const int N = 1e6 + 10;
int q[N];

void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;
    int i = l - 1,j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i++; while (q[i] < x);
        do j--; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

int main ()
{
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    quick_sort(q, 0, n - 1);
    for (int i = 0; i < n; i++) printf("%d ", q[i]);
    return 0;
}
```

## 归并排序

> 1. 确定分界点
> 2. 递归排序 left、right
> 3. left、right 排序结果合二为一

### 归并模版

```C++
void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);

    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k ++ ] = q[i ++ ];
        else tmp[k ++ ] = q[j ++ ];

    while (i <= mid) tmp[k ++ ] = q[i ++ ];
    while (j <= r) tmp[k ++ ] = q[j ++ ];

    for (i = l, j = 0; i <= r; i ++, j ++ ) q[i] = tmp[j];
}
```

### AcWing 787. 归并排序

```C++
#include <iostream>

using namespace std;

int const N = 1e6 + 10;
int q[N];
int tmp[N];
int n;

void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
    {
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];
    for (i = l, j = 0; i <= r; i++, j++) q[i] = tmp[j];
}

int main ()
{
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    merge_sort(q, 0, n - 1);
    for (int i = 0; i < n; i++) printf("%d ", q[i]);
    return 0;
}
```

## 二分

### 二分模版

- 整数二分

```C++
bool check(int x) {/* ... */} // 检查 x 是否满足某种性质

int bsearch_1(int l, int r)
{
	while (l < r)
	{
		int mid = l + r >> 1;
		if (check(mid)) r= mid;
		else l = mid + 1;
	}
	return l;
}

int bsearch_2(int l, int r)
{
	while (l < r)
	{
		int mid = l + r + 1 >> 1;
		if (check(mid)) l = mid;
		else r = mid - 1;
	}
	return l;
}
```

- 小数二分

```C++
bool check(int x) {/.../}

int bsearch_3(int l, int r)
{
	// 保留 N 位，r - l > 1e + (N + 2)
	int const esp = 1e-6;
	while (r - l > esp)
	{
		int mid = (l + r) / 2;
		if (check(mid)) r= mid;
		else l= mid;
	}
	return l;
}
```

### AcWing 789. 数的范围

```C++
#include <iostream>

using namespace std;

int const N = 100010;
int q[N];
int n, k;

void bsearch()
{
    int x, l = 0, r = n - 1;
    scanf("%d", &x);

    while (l < r)
    {
        int mid = l + r >> 1;
        if (q[mid] >= x) r = mid;
        else l = mid + 1;
    }

    if (q[l] != x)
    {
        cout << "-1 -1" << endl;
        return;
    }
    cout << l << " ";
    r = n - 1;
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (q[mid] <= x) l = mid;
        else r = mid - 1;
    }
    cout << l << endl;
}

int main()
{
    scanf("%d%d", &n, &k);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    for (int i = 0; i < k; i++) bsearch();
}
```

### 790. 数的三次方根

```C++
#include <iostream>

using namespace std;

int main()
{
    double x;
    cin >> x;
    double l = min(-x, x), r = max(-x, x);

    while (r - l > 1e-8)
    {
        double mid = (l + r) / 2;
        if (mid * mid * mid >= x) r = mid;
        else l = mid;
    }
    printf("%lf", l);
    return 0;
}
```
